import pandas as pd
import pyspark.sql.functions as F

from pyspark.sql.window import Window
from plotnine import *

dat = spark.table("safegraph.monthly_patterns")
poi = spark.table("safegraph.core_places")

# datasets
spark.sql("CREATE DATABASE maverik_challenge")
poi_mav = poi.filter(F.col("brand") == "SG_BRAND_857fcc0687933d1589554945329fe236")
poi_mav.repartition(5).write.saveAsTable("maverik_challenge.core_places")

cols = ("city","location_name","parent_placekey", "postal_code", "region", "street_address")
dat_mav = dat.join(poi_mav.drop(*cols), ['placekey'], "inner")
dat_mav.repartition(50).write.saveAsTable("maverik_challenge.monthly_patterns")
